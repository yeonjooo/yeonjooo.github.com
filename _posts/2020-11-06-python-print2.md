## print
- format
	- print("{} and {}".format('you', 'me'))
	- print("{0} and {1} and {0}".format('you', 'me'))
		- 0 : you
		- 1 : me
	- print("{a} and {b}".format(a='you', b='me'))
	- print("%s's the most favorite number is %d" **%** ('yeonju', 28))
	- print("%s's the most favorite number is %d" **%** ('yeonju', 28))
